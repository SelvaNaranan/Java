import java.util.*;
class Main {
    public static void main(String[] args) {
        System.out.println("Enter Range: ");
        Scanner sc=new Scanner(System.in);
        int a=sc.nextInt();
        int ini=0,corress=1;
        for(int i=0;i<=a;i++){
            System.out.print(ini+" ");
            int add=ini+corress;
           ini=corress;
           corress=add;
        }
    }
}
